class DatabaseMCPServer(MCPServer):
    def __init__(self):
        super().__init__("database-server", "1.0.0")
        self.db_path = "example.db"

    @Tool("query_database")
    async def query_database(self, sql: str) -> dict:
        """æ‰§è¡ŒSQLæŸ¥è¯¢å¹¶è¿”å›ç»“æœ"""
        try:
            conn = sqlite3.connect(self.db_path)
            cursor = conn.cursor()
            cursor.execute(sql)
            results = cursor.fetchall()
            columns = [desc[0] for desc in cursor.description]

            return {
                "success": True,
                "data": [dict(zip(columns, row)) for row in results],
                "count": len(results)
            }
        except Exception as e:
            return {
                "success": False,
                "error": str(e)
            }
        finally:
            conn.close()

    async def get_user_data(self) -> str:
        """è·å–ç”¨æˆ·æ•°æ®èµ„æº"""
